main.cpp
```cpp
#include "widgets.h"
#include <QApplication>//应用程序头文件
/*
 *QT编辑器快捷键
 * 注释   Ctrl+/
 * 运行   Ctrl+r
 * 构建   Ctrl+b
 * 转到定义  F2
 * 返回Alt+left键
 * 前进Alt+right键
 */
int main(int argc, char *argv[])
{
    QApplication a(argc, argv);//定义应用程序对象
    widgets w;//定义空窗口对象
    w.show();//调用空窗口子函数   显示窗口

    return a.exec();//调用应用程序对象消息循环函数
}

.h

#ifndef WIDGETS_H
#define WIDGETS_H//宏定义

#include <QMainWindow>//包含Qt空窗口头文件

namespace Ui {
class widgets;
}

class widgets : public QMainWindow
{
    Q_OBJECT//支持信号  和  槽

public:
    explicit widgets(QWidget *parent = nullptr);//构造函数
    ~widgets();//析构函数

private:
    Ui::widgets *ui;
};

#endif // WIDGETS_H
